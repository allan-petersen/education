FIRST
=====

-- Extract the first non-NULL eanlocation from the path
REGEXP_SUBSTR(SYS_CONNECT_BY_PATH(sor.eanlocation, '|'),
                                                      '[^|]+', -- Match any sequence of non | characters
                                                      1, -- Start from the first character of the path
                                                      1 -- Extract only the first match (non-NULL eanlocation)
)

	  -- Extract the first non-NULL eanlocation from the path
      TO_NUMBER(                                             
         RTRIM(                                          
            SUBSTR(                                         
               -- Remove any leading pipe '|' from the generated path
               LTRIM(sys_connect_by_path(sor.eanlocation, '|'), '|'),
               0,
               -- Find the position of the first delimiter in the path
               INSTR(                                        
                  -- Append a trailing pipe '|' to ensure a delimiter exists, 
                  -- then remove any leading '|' from the concatenated string
                  LTRIM(sys_connect_by_path(sor.eanlocation, '|') || '|', '|'),
                  '|', 
                  1                                       -- Start search at position 1
                ) + 1 --INSTR end. +1 to move past the delimiter
            ), '|' -- SUBSTR functionend. Remove any trailing delimiter
         ) -- RTRIM end
      ) -- TO_NUMBER end
      AS inheritedLocationNumberId,   


	  
LAST
====	  

-- Extract the last non-NULL eanlocation from the path
REGEXP_SUBSTR(SYS_CONNECT_BY_PATH(sor.eanlocation, '|'),
                                                      '([^|]+)[|]*$', -- Match one or more characters that are not |. Match any trailing |. $ forces the match to occur at the end of the string.
                                                      1, -- Start from the first character of the path
                                                      1, -- Extract only the first match (non-NULL eanlocation)
                                                      NULL,
                                                      1 -- Return the first group only          
)

      -- Extract the last non-NULL eanlocation from the path
      TO_NUMBER(                                             
         SUBSTR(                                            
            LTRIM(sys_connect_by_path(sor.eanlocation, '|'), '|'),   -- Removes leading delimiter from the path
            INSTR(                                           
               LTRIM(sys_connect_by_path(sor.eanlocation, '|'), '|') || '|', -- concatenated with a trailing '|'
               '|',                                        -- Delimiter to search for in INSTR
               -1                                          -- Start searching from the end of the string (last occurrence)
            ) + 1,                                         -- INSTR end.  +1 to move past the last delimiter
         )                                                 -- SUBSTR end
      ) --  TO_NUMBER end
      AS inheritedLocationNumberId,
